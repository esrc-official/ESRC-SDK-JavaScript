class APIClient {
    /**
     * Validate application ID.
     * @param {strubg} appId Application ID.
     * @param {commonCallback} callback Callback handler.
     * @static
     */
    static async validateLicense(appId, callback) {
        var auth = false;
    
        // const url = "http://3.36.153.168:80/api/v1/applications/auth";
        const url = "http://api-esrc.co.kr:80/api/v1/applications/auth";
        fetch(url, {
            method: "POST",
            headers: {
                "Content-Type": "application/json",
                "Accept": "application/json"
            },
            body: JSON.stringify({
                "app_id": appId,
                "platform": "javascript",
                "name": "ESRC",
                "package": "com.esrc.sdk.javascript",
                "version": "0.1.1",
                "product": "esrc",
                "device_id": ""
            })
        }).then((response) => {
            if(response.ok){
                response.json().then(function(json) {
                    auth = json.auth;
                    callback(auth);
                });
            } else {
                console.log("response failed");
                auth = false;
                callback(auth);
            }
        }).catch((error) => {
            auth = false;
            callback(auth);
        });
    }
}
