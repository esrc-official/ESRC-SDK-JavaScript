const {ESRCRemoteHR, ESRCHRV, ESRCEngagement } = require("./../ESRCType.js");
const Heart = require("./../native/sdk/Heart.js");

let instance = null;

/**
 * EngagementRecognizer class.
 * 
 * @author Hyunwoo Lee <lhw4846@esrc.co.kr>
 * @license Copyright (c) 2022 Emotion Science Research Center all rights reserved.
 */
class EngagementRecognizer {
    /**
     * @property {onRecognizedEngagementCallback} onRecognizedEngagementCallback A callback for when engagement is recognized.
     */
    onRecognizedEngagementCallback;
    /**
     * @property {boolean} enableEngagement Whether recognize engagement or not.
     */
    enableEngagement = false;

    /**
     * EngagementRecognizer constructor.
     */
    constructor() {
        if (instance) {
            return instance;
        }
        instance = this;
    }

    /**
     * Initializes engagement recognizer
     * @param {boolean} enableEngagement Whether recognize engagement or not.
     * @param {onRecognizedEngagementCallback} onRecognizedEngagementCallback A callback for when engagement is recognized.
     * @returns {boolean} Returns true if the initialization succeeded. Otherwise, false.
     */
    initialize(enableEngagement, onRecognizedEngagementCallback) {
        var ret = true;

        // Set property
        this.enableEngagement = enableEngagement;

        // Set handler
        this.onRecognizedEngagementCallback = onRecognizedEngagementCallback ?? undefined;

        // Initialize variables
        this.emotionScore = 0;
        this.emotionProbs = Array(ESRCEngagement.EMOTION_COUNT).fill(0);
        this.emotion = ESRCEngagement.EMOTION_NONE;

        // Initialize engagement recognition task
        ret = Heart.initEngagementRecognitionTask(this.onRecognizedEngagement)

        return ret;
    }

    /**
     * Releases engagement recognizer.
     * @returns {boolean} Returns true if the release succeeded. Otherwise, false.
     */
    release() {
        var ret = true;

        // Release engagement recognition task
        ret = Heart.releaseEngagementRecognitionTask();

        return ret;
    }

    /**
     * Feed HR or HRV on engagement recognition task.
     * @param {ESRCRemoteHR|ESRCHRV} arg ESRCRemoteHR or ESRCHRV instance.
     */
    feed(arg) {
        if (this.enableEngagement) {
            if (arg instanceof ESRCRemoteHR) {
                // Feed HR engagement recognition task
                Heart.feedOnEngagementRecognitiontInTimeDomainTask(arg);
            } 
            else if (arg instanceof ESRCHRV) {
                // Feed HRV on engagement recognition task
                Heart.feedOnEngagementRecognitiontInFreqDomainTask(arg);
            }
        }
    }

    /**
     *  A callback for when engagement is recognized from native.
     * @param {number} id Identification.
     * @param {number} score Score of emotion.
     * @param {Array.<number>} probs Probabilities of emotions.
     * @param {number} emotion Emotion.
     */
    onRecognizedEngagement(id, score, probs, emotion) {
        // Convert native to ESRCType
        const engagement = new ESRCEngagement(id, score, probs, emotion);

        // Callback
        if (EngagementRecognizer.getInstance().onRecognizedEngagementCallback !== undefined) {
            EngagementRecognizer.getInstance().onRecognizedEngagementCallback(engagement);
        }
    }

    /**
     * EngagementRecognizer instance.
     * @returns {EngagementRecognizer} Returns EngagementRecognizer instance.
     */
    static getInstance() {
        return new EngagementRecognizer();
    }
}
