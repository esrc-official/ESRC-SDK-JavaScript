class FileManager {
    /**
     * Create file from url.
     * @param {string} path String to access loaded file through cv.
     * @param {string} url Path of the actual file on FS.
     * @param {onloadCallback} callback Callback handler.
     * @static
     */
    static createOpenCVFileFromUrl = function(path, url, callback) {
        let request = new XMLHttpRequest();
        request.open("GET", url, true);
        request.responseType = "arraybuffer";
        request.onload = function(ev) {
            if (request.readyState === 4) {
                let data = new Uint8Array(request.response);
                cv.FS_createDataFile("/", path, data, true, false, false);
                callback();
            } else {
                console.log("Failed to liad " + url);
            }
        };
        request.send();
    }
}

module.exports = FileManager;
